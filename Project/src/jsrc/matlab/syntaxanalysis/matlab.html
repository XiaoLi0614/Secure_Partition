File ::= ( FunctionHeader )? ( Statement )* <EOF>
FunctionHeader ::= "specification" ( OutputClause )? Identifier "(" ( IdentifierList )? ")" ( ";" )?

WhileSt ::= "while" Expression ( Statement )* "end"
SwitchSt ::= "switch" Expression ( Case )* "otherwise" ( Statement )* "end"
